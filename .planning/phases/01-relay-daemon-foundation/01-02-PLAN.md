    **src/config.ts:**
    ```typescript
    import { z } from "zod";

    const ConfigSchema = z.object({
      port: z.coerce.number().default(3000),
      relaySecret: z.string().min(1),  // used for optional HMAC validation in future
      ttlSeconds: z.coerce.number().default(300),  // 5 min peer TTL
      storageType: z.enum(["sqlite", "memory"]).default("sqlite"),
      dbPath: z.string().default("./relay.db"),
      allowedOrigins: z.string().default("*"),
    });

    export type RelayConfig = z.infer<typeof ConfigSchema>;

    export function loadConfig(): RelayConfig {
      const result = ConfigSchema.safeParse({
        port: process.env["PORT"],
        relaySecret: process.env["RELAY_SECRET"],
        ttlSeconds: process.env["RELAY_TTL_SECONDS"],
        storageType: process.env["RELAY_STORAGE"] ?? "sqlite",
        dbPath: process.env["RELAY_DB_PATH"],
        allowedOrigins: process.env["RELAY_ALLOWED_ORIGINS"],
      });
      if (!result.success) {
        const missing = result.error.errors.map(e => e.path.join(".")).join(", ");
        throw new Error(`Relay config invalid. Missing required: ${missing}`);
      }
      return result.data;
    }
    ```

  <action>
    **src/app.ts** — factory function, not a singleton:
    ```typescript
    import express from "express";
    import type { PeerStore } from "./store/index.js";
    import type { RelayConfig } from "./config.js";
    import { registerRouter } from "./routes/register.js";
    import { lookupRouter } from "./routes/lookup.js";
    import { httpsOnlyMiddleware } from "./middleware/httpsOnly.js";
    import { rateLimitMiddleware } from "./middleware/rateLimit.js";

    export function createApp(config: RelayConfig, store: PeerStore) {
      const app = express();
      app.use(express.json());
      app.use(httpsOnlyMiddleware);
      app.use(rateLimitMiddleware({ maxRequests: 100, windowMs: 60_000 }));
      app.use("/register", registerRouter(store, config));
      app.use("/lookup", lookupRouter(store, config));
      app.get("/health", (_req, res) => { res.json({ ok: true }); });
      return app;
    }
    ```

    **src/middleware/httpsOnly.ts** — reject requests where `req.secure === false` AND `req.headers["x-forwarded-proto"] !== "https"`. In development (NODE_ENV=development), skip this check.

    **src/middleware/rateLimit.ts** — simple in-memory Map-based rate limiter. No external dependency (express-rate-limit is optional; hand-roll for zero deps). Clears expired windows on each request.

    **src/routes/register.ts** — validates body with Zod (publicKey: base64 string 44 chars, endpoint: string, timestampMs: number). Calls store.upsert(). Returns RegisterResponse.

    **src/routes/lookup.ts** — reads `:publicKey` param. Calls store.findByPublicKey(). Returns LookupResponse or RelayError 404.

    All responses use `@homelan/shared` types. Error responses always match `RelayError` schema.
