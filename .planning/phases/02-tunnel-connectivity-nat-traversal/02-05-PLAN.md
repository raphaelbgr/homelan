export type ConnectionState = "idle" | "connecting" | "connected" | "disconnecting" | "error";
export type TunnelMode = "full-gateway" | "lan-only";
export interface DaemonStatus {
  state: ConnectionState;
  mode: TunnelMode | null;
  latencyMs: number | null;
  throughputBytesPerSec: number | null;
  hostInfo: HostInfo | null;
  connectedPeers: PeerInfo[];

- `homelan status [--human] [--json]`
- CLI shows real-time spinner/progress during connect (never hangs silently)

CLI library: Commander.js 12.x (already in tech stack decision).

    Create packages/cli/src/commands/connect.ts:
    - connectCommand(): Command (Commander.js Command object)
    - Options: --mode <mode> (default "lan-only"), --timeout <seconds> (default 30), --retry <count> (default 0), --json
    - Implementation:
      1. Check daemon running via ipcClient.isRunning() — if false, console.error + process.exit(3)
      2. Start ora spinner: "Discovering peer..."
      3. If --json flag: subscribe to SSE /events to get connection_progress events, print each as JSON line
      4. If not --json: update spinner text as progress events arrive
      5. POST /connect { mode } with fetch (streaming — or just POST and poll /status for state change)
      6. On success (state → "connected"): spinner.succeed("Connected"), exit 0
      7. On timeout (--timeout seconds elapsed): spinner.fail("Timeout"), exit 2
      8. On error response: spinner.fail(message), exit 1
      9. --retry: retry up to N times on error/timeout before exiting

    For simplicity: POST /connect (daemon does all work asynchronously) → then poll GET /status every 500ms until state is "connected" or "error" or timeout. This avoids SSE parsing complexity in the CLI. SSE is for GUI (Phase 4).
